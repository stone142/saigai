// ヘルパー関数: 指数バックオフを使用した非同期呼び出し
const withRetry = async (fn, retries = 3) => {
    for (let i = 0; i < retries; i++) {
        try {
            return await fn();
        } catch (error) {
            if (i < retries - 1) {
                const delay = Math.pow(2, i) * 1000 + Math.random() * 1000;
                await new Promise(resolve => setTimeout(resolve, delay));
            } else {
                throw error;
            }
        }
    }
};

/**
 * 8桁のゼロ埋め患者IDを生成（正規化）
 * 例: "123" -> "00000123"
 * @param {string|number} patientId
 */
const formatPatientId = (patientId) => {
    const idString = String(patientId).replace(/[^0-9]/g, ''); // 数字以外を削除
    // 8桁になるようにゼロ埋め
    return idString.padStart(8, '0');
};

/**
 * タイムスタンプを相対時間で表示
 * @param {object} timestamp - Firestore Timestamp object
 */
const formatRelativeTime = (timestamp) => {
    if (!timestamp || !timestamp.toDate) return '不明';
    const now = new Date();
    const then = timestamp.toDate();
    const seconds = Math.floor((now - then) / 1000);

    if (seconds < 60) return `${seconds}秒前に更新`;
    const minutes = Math.floor(seconds / 60);
    if (minutes < 60) return `${minutes}分前に更新`;
    const hours = Math.floor(minutes / 60);
    if (hours < 24) return `${hours}時間前に更新`;
    const days = Math.floor(hours / 24);
    return `${days}日前に更新`;
};

/**
 * データをCSV形式に変換してダウンロード
 * @param {Array<Object>} data - 傷病者リスト
 */
const exportToCsv = (data) => {
    if (data.length === 0) {
        // alert()の代わりにカスタムモーダルまたはメッセージを使用
        console.warn("エクスポートするデータがありません。");
        return;
    }

    const headers = [
        "患者ID", "トリアージ番号", "エリア区分", "氏名", "性別", "年齢",
        "傷病名", "処置状況", "転帰", "登録日時", "最終更新日時",
    ];
    const headerRow = headers.join(',');
    
    const rows = data.map(item => {
        const registeredAt = item.registeredAt?.toDate().toLocaleString('ja-JP', { timeZone: 'Asia/Tokyo' }) || '';
        const lastUpdated = item.lastUpdated?.toDate().toLocaleString('ja-JP', { timeZone: 'Asia/Tokyo' }) || '';

        // CSVフィールド内のダブルクォートをエスケープ (2つのダブルクォートに変換)
        const escapeCsv = (text) => `"${(text || '').replace(/"/g, '""')}"`;

        const row = [
            escapeCsv(formatPatientId(item.id)), // IDをゼロ埋めしてエクスポート
            escapeCsv(item.triageNumber),
            escapeCsv(item.areaDivision),
            escapeCsv(item.name),
            escapeCsv(item.gender),
            escapeCsv(item.age),
            escapeCsv(item.injuryName),
            escapeCsv(item.treatmentStatus),
            escapeCsv(item.outcome),
            escapeCsv(registeredAt),
            escapeCsv(lastUpdated),
        ].join(',');
        return row;
    });

    // \ufeff はBOM (UTF-8) で、Excel等での文字化けを防ぐ
    const csvContent = "\ufeff" + headerRow + '\n' + rows.join('\n'); 

    const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
    const link = document.createElement('a');
    if (link.download !== undefined) {
        const url = URL.createObjectURL(blob);
        link.setAttribute('href', url);
        link.setAttribute('download', `CasualtyList_${new Date().toISOString().slice(0, 10)}.csv`);
        link.style.visibility = 'hidden';
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
    }
};

        if (name === 'triageNumber' || name === 'areaDivision') {

/**
 * 傷病者ID入力画面
 */
const PatientIdEntry = ({ onIdSubmit }) => {
    // ユーザーが入力したそのままの値を保持
    const [patientIdInput, setPatientIdInput] = useState(''); 
    const [error, setError] = useState('');

    const handleInput = (e) => {
        // 数字以外を削除
        const value = e.target.value.replace(/[^0-9]/g, '');
        setPatientIdInput(value);
        
        // リアルタイムバリデーション（最大8桁）
        if (value.length > 8) {
             setError('患者IDは最大8桁の数字で入力してください。');
        } else if (value.length === 0) {
            setError('患者IDを入力してください。');
        } else {
            setError('');
        }
    };

    const handleSubmit = (e) => {
        e.preventDefault();
        const rawId = patientIdInput.trim();

        if (rawId.length === 0) {
            setError('患者IDを入力してください。');
            return;
        }
        if (rawId.length > 8) {
            setError('患者IDは最大8桁の数字で入力してください。');
            return;
        }

        // 入力されたIDを必ず8桁ゼロ埋め形式に正規化して送信
        const formattedId = formatPatientId(rawId);
        onIdSubmit(formattedId);
    };

    const inputClassName = `w-full p-4 text-3xl text-center tracking-widest border rounded-lg focus:ring-4 focus:ring-blue-100 transition duration-150 ${error ? 'border-red-500' : 'border-gray-300 focus:border-blue-500'}`;

    return (
        <div className="bg-white p-6 rounded-xl shadow-2xl max-w-lg mx-auto">
            <h2 className="text-2xl font-bold text-gray-800 mb-4 border-b pb-2">
                現場入力モード
            </h2>
            <p className="text-gray-600 mb-6">
                患者ID (最大8桁の数字) を入力してください。<br/>
                <span className="text-xs text-gray-500">(例: 123 と 00000123 は同じIDとして扱われます)</span>
            </p>
            <form onSubmit={handleSubmit} className="space-y-4">
                <div>
                    <label htmlFor="patientId" className="block text-sm font-medium text-gray-700 mb-2">
                        患者ID (8桁)
                    </label>
                    <input
                        id="patientId"
                        type="tel" // モバイルでの数字入力を容易にする
                        value={patientIdInput}
                        onChange={handleInput}
                        placeholder="例: 123 または 00000123"
                        className={inputClassName}
                        maxLength="8" // HTML側の最大桁数制限
                    />
                    {error && <p className="text-red-500 text-sm mt-2">{error}</p>}
                </div>
                <button
                    type="submit"
                    className="w-full bg-blue-600 hover:bg-blue-700 text-white font-bold py-3 px-4 rounded-lg shadow-lg transition duration-150 disabled:bg-blue-400"
                    disabled={error.length > 0 || patientIdInput.length === 0}
                >
                    詳細入力へ進む
                </button>
            </form>
        </div>
    );
};

        }

        // 2. 検索フィルタ (ID, 氏名, 傷病名, 処置, トリアージ番号)
        if (searchTerm) {
            const lowerCaseSearch = formatPatientId(searchTerm).toLowerCase(); // 検索語もゼロ埋め正規化
            const lowerCaseTriageSearch = searchTerm.toLowerCase();

            list = list.filter(c => 
                // IDはFirestoreに保存されているゼロ埋めIDと検索語(ゼロ埋め済み)を比較
                c.id.includes(lowerCaseSearch) ||
                // トリアージ番号、氏名、傷病名、処置はそのままの文字列で検索
                c.triageNumber?.toLowerCase().includes(lowerCaseTriageSearch) ||
                c.name?.toLowerCase().includes(lowerCaseTriageSearch) ||
                c.injuryName?.toLowerCase().includes(lowerCaseTriageSearch) ||
                c.treatmentStatus?.toLowerCase().includes(lowerCaseTriageSearch)
            );
        }
        
        // 3. ソート (最終更新日時で降順)
        return list.sort((a, b) => {
            const timeA = a.lastUpdated?.toDate().getTime() || 0;
            const timeB = b.lastUpdated?.toDate().getTime() || 0;
            return timeB - timeA;
        });
    }, [casualties, filter, searchTerm]);

    const AreaTag = ({ categoryId }) => {
        const category = AREA_CATEGORIES.find(c => c.id === categoryId);
        if (!category) return <span className="text-gray-500 text-sm">不明</span>;
        return (
            <span className={`px-2 py-1 text-xs font-bold rounded-full ${category.color} ${category.text}`}>
                {category.id}
            </span>
        );
    };

    return (
        <div className="bg-white p-4 md:p-8 rounded-xl shadow-2xl w-full">
            <h2 className="text-2xl md:text-3xl font-bold text-gray-800 mb-6 flex items-center justify-between border-b pb-3">
                <span className="flex items-center"><Users className="w-6 h-6 mr-3 text-red-600" />本部 状況把握リスト</span>
                <button 
                    onClick={onSwitchMode}
                    className="flex items-center bg-gray-200 text-gray-700 hover:bg-gray-300 px-4 py-2 rounded-lg text-sm font-semibold transition duration-150"
                >
                    <Edit3 className="w-4 h-4 mr-2" />
                    現場入力モードへ
                </button>
            </h2>

            <div className="flex flex-col md:flex-row space-y-3 md:space-y-0 md:space-x-4 mb-6">
                {/* フィルタボタン */}
                <div className="flex space-x-2 overflow-x-auto p-1 bg-gray-50 rounded-lg shadow-inner flex-shrink-0">
                    {['全て', ...AREA_CATEGORIES.map(c => c.id)].map(t => (
                        <button
                            key={t}
                            onClick={() => setFilter(t)}
                            className={`px-4 py-2 rounded-lg font-semibold text-sm transition duration-150 whitespace-nowrap ${
                                filter === t
                                    ? (t === '全て' ? 'bg-blue-600 text-white shadow-lg' : AREA_CATEGORIES.find(c => c.id === t)?.color + ' text-white shadow-lg')
                                    : 'bg-white text-gray-700 hover:bg-gray-100'
                            }`}
                        >
                            {t}
                        </button>
                    ))}
                </div>

                {/* 検索とエクスポート */}
                <div className="flex flex-grow space-x-2">
                    <div className="relative flex-grow">
                        <input
                            type="text"
                            placeholder="ID(0埋め), トリアージ番号, 氏名などで検索..."
                            value={searchTerm}
                            onChange={(e) => setSearchTerm(e.target.value)}
                            className="w-full p-3 pl-10 border border-gray-300 rounded-lg focus:ring-2 focus:ring-indigo-100 focus:border-indigo-500"
                        />
                        <Search className="w-5 h-5 text-gray-400 absolute left-3 top-1/2 transform -translate-y-1/2" />
                    </div>
                    
                    <button
                        onClick={() => exportToCsv(casualties)}
                        className="flex items-center bg-green-600 hover:bg-green-700 text-white font-semibold py-3 px-4 rounded-lg shadow-md transition duration-150 flex-shrink-0"
                    >
                        <Download className="w-5 h-5 mr-2" />
                        CSVエクスポート
                    </button>
                </div>
            </div>

            {/* 傷病者リストのテーブル */}
            <div className="overflow-x-auto rounded-lg border border-gray-200">
                <table className="min-w-full divide-y divide-gray-200">
                    <thead className="bg-gray-50">
                        <tr>
                            <th className="px-3 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">ID</th>
                            <th className="px-3 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">トリアージ番号</th>
                            <th className="px-3 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">エリア</th>
                            <th className="px-3 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">氏名/年齢</th>
                            <th className="px-3 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">傷病名/処置</th>
                            <th className="px-3 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">転帰</th>
                            <th className="px-3 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">最終更新</th>
                        </tr>
                    </thead>
                    <tbody className="bg-white divide-y divide-gray-100">
                        {filteredCasualties.length > 0 ? (
                            filteredCasualties.map(c => (
                                <tr key={c.id} className="hover:bg-yellow-50">
                                    <td className="px-3 py-3 whitespace-nowrap text-sm font-medium text-gray-900">{c.id}</td> {/* IDはFirestoreに保存されているゼロ埋めID */}
                                    <td className="px-3 py-3 whitespace-nowrap text-sm text-gray-700">{c.triageNumber || '---'}</td>
                                    <td className="px-3 py-3 whitespace-nowrap">
                                        <AreaTag categoryId={c.areaDivision} />
                                    </td>
                                    <td className="px-3 py-3 text-sm text-gray-700">
                                        <div className="font-semibold">{c.name || '---'}</div>
                                        <div className="text-xs text-gray-500">{c.gender || '不明'}, {c.age || '年齢不明'}</div>
                                    </td>
                                    <td className="px-3 py-3 text-sm text-gray-700 max-w-xs truncate">
                                        <div className="font-medium text-gray-800">{c.injuryName || '傷病名未入力'}</div>
                                        <div className="text-xs text-blue-600 truncate">{c.treatmentStatus || '処置未入力'}</div>
                                    </td>
                                    <td className="px-3 py-3 text-sm text-gray-700 max-w-xs truncate">
                                        {c.outcome || '---'}
                                    </td>
                                    <td className="px-3 py-3 whitespace-nowrap text-sm text-gray-500">
                                        <div className="flex items-center text-xs">
                                            <Clock className="w-3 h-3 mr-1" />
                                            {formatRelativeTime(c.lastUpdated)}
                                        </div>
                                    </td>
                                </tr>
                            ))
                        ) : (
                            <tr>
                                <td colSpan="7" className="px-6 py-4 text-center text-gray-500">
                                    該当する傷病者は見つかりませんでした。
                                </td>
                            </tr>
                        )}
                    </tbody>
                </table>
            </div>
            <div className="mt-4 text-sm text-gray-600">
                表示件数: {filteredCasualties.length} / 全件: {casualties.length}
            </div>
        </div>
    );
};


/**
 * メインアプリケーションコンポーネント
 */
const App = () => {
    const [mode, setMode] = useState('entry'); // 'entry' or 'hq'
    const [db, setDb] = useState(null);
    const [userId, setUserId] = useState(null);
    const [isAuthReady, setIsAuthReady] = useState(false);
    const [currentPatientId, setCurrentPatientId] = useState(null);
    const [currentCasualtyData, setCurrentCasualtyData] = useState(null);
    const [casualties, setCasualties] = useState([]);
    const [loading, setLoading] = useState(true);

    // 1. Firebase初期化と認証
    useEffect(() => {
        try {
            const app = initializeApp(firebaseConfig);
            const firestore = getFirestore(app);
            const auth = getAuth(app);
            setDb(firestore);

            const unsubscribe = onAuthStateChanged(auth, async (user) => {
                if (user) {
                    setUserId(user.uid);
                    setIsAuthReady(true);
                } else {
                    try {
                        if (initialAuthToken) {
                            await signInWithCustomToken(auth, initialAuthToken);
                        } else {
                            await signInAnonymously(auth);
                        }
                    } catch (error) {
                        console.error("Authentication failed:", error);
                        // Fallback to anonymous if custom token fails
                        await signInAnonymously(auth);
                    }
                }
            });
            return () => unsubscribe();
        } catch (error) {
            console.error("Firebase initialization failed:", error);
            setLoading(false);
            setIsAuthReady(true);
        }
    }, []);

    // 2. 傷病者リストのリアルタイム監視 (HQ Mode用)
    useEffect(() => {
        if (!db || !userId || !isAuthReady) return;

        // 公開コレクションへのパス
        const casualtiesColRef = collection(db, 'artifacts', appId, 'public', 'data', 'casualties');
        const q = query(casualtiesColRef);

        const unsubscribe = onSnapshot(q, (snapshot) => {
            const list = snapshot.docs.map(doc => ({
                id: doc.id,
                ...doc.data(),
            }));
            setCasualties(list);
            setLoading(false);
        }, (error) => {
            console.error("Error fetching casualty data:", error);
            setLoading(false);
        });

        return () => unsubscribe();
    }, [db, userId, isAuthReady]);

    // 3. 患者IDが入力されたときの処理
    const handleIdSubmit = useCallback(async (formattedId) => {
        // formattedIdは既に8桁のゼロ埋め文字列 (例: "00000123")
        setCurrentPatientId(formattedId);

        if (db) {
            // FirestoreのドキュメントIDとしてゼロ埋めIDを使用
            const casualtyRef = doc(db, 'artifacts', appId, 'public', 'data', 'casualties', formattedId);
            const docSnap = await getDoc(casualtyRef);

            if (docSnap.exists()) {
                // 既存データをロード
                setCurrentCasualtyData(docSnap.data());
            } else {
                // 新規データを作成 (初期値設定)
                setCurrentCasualtyData({
                    // IDはドキュメントIDと同じく、ゼロ埋めされたものをそのまま使用
                    id: formattedId,
                    triageNumber: '', // 自由記載
                    areaDivision: '',
                    name: '',
                    gender: '',
                    age: '',
                    injuryName: '',
                    treatmentStatus: '',
                    outcome: '', // 転帰
                });
            }
        }
    }, [db]);

    // 4. データ保存後の処理
    const handleSaveComplete = (savedId) => {
        // ID入力画面に戻る
        setCurrentPatientId(null);
        setCurrentCasualtyData(null);
    };

    // 5. モード切り替え
    const switchMode = () => {
        setMode(prev => prev === 'entry' ? 'hq' : 'entry');
        setCurrentPatientId(null);
        setCurrentCasualtyData(null);
    };

    if (loading || !isAuthReady) {
        return (
            <div className="min-h-screen flex items-center justify-center bg-gray-100">
                <div className="text-center p-6 bg-white rounded-lg shadow-xl">
                    <svg className="animate-spin h-8 w-8 text-blue-600 mx-auto mb-3" viewBox="0 0 24 24">
                        <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="4"></circle>
                        <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>
                    </svg>
                    <p className="text-gray-700">システム準備中...</p>
                    <p className="text-xs text-gray-400 mt-2">ユーザーID: {userId}</p>
                </div>
            </div>
        );
    }
    
    return (
        <div className="min-h-screen bg-gray-100 p-4 md:p-8 font-sans">
            <header className="text-center mb-8">
                <h1 className="text-4xl font-extrabold text-gray-900 flex items-center justify-center">
                    <img 
                        src="https://placehold.co/40x40/ef4444/ffffff?text=D" 
                        alt="Disaster Logo" 
                        className="rounded-full mr-3 shadow-md"
                        onError={(e) => { e.target.onerror = null; e.target.src = "https://placehold.co/40x40/ef4444/ffffff?text=D" }} 
                    />
                    災害医療 傷病者管理システム
                </h1>
                <p className="text-sm text-gray-500 mt-1">
                    ユーザーID: <span className="font-mono bg-gray-200 p-1 rounded text-xs">{userId}</span>
                </p>
            </header>

            <main>
                {mode === 'entry' ? (
                    currentPatientId && currentCasualtyData ? (
                        <CasualtyDetailForm
                            db={db}
                            casualty={currentCasualtyData}
                            setCasualty={setCurrentCasualtyData}
                            onSave={handleSaveComplete}
                            onBack={() => {
                                setCurrentPatientId(null);
                                setCurrentCasualtyData(null);
                            }}
                            userId={userId}
                        />
                    ) : (
                        <div className="space-y-4">
                            <PatientIdEntry onIdSubmit={handleIdSubmit} />
                            <div className="text-center">
                                <button
                                    onClick={switchMode}
                                    className="flex items-center mx-auto text-blue-600 hover:text-blue-700 font-semibold py-2 px-4 rounded-lg transition duration-150"
                                >
                                    <Eye className="w-5 h-5 mr-2" />
                                    本部モード（リスト閲覧）に切り替える
                                </button>
                            </div>
                        </div>
                    )
                ) : (
                    <HeadquartersView
                        casualties={casualties}
                        db={db}
                        userId={userId}
                        onSwitchMode={switchMode}
                    />
                )}
            </main>
        </div>
    );
};

export default App;
